is_dwarves = {
	OR = {
		original_tag = ERE
		AND = {
			original_tag = MOA
			NOT = {
				OR = {
					has_government = belligerent
					has_government = revolutionary
				}
			}
		}
	}
}

# is the country is dwarven at the start?
is_dwarves_init = {
	OR = {
		original_tag = ERE
		original_tag = MOA
	}
}
