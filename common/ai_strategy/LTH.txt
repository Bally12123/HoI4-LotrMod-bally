    enable = {
		OR = {
			AND = {
				has_government = unaligned
				75 = { check_variable = { industrial_complex_level < 14 } } # build more if unaligned
			}
			75 = { check_variable = { industrial_complex_level < 12 } }
		}
		has_war = no
    }
    allowed = {
        original_tag = LTH
    }

    abort_when_not_enabled = yes

    ai_strategy = {
		type = build_building
		id = industrial_complex
		target = 75
		value = 15 # Get *at least* 12 civs into state 75 first, that way we can get the most out of the "twin hearts" focus
	}
}
