is_elves = {
	OR = {
		original_tag = LIN
		original_tag = LTH
		original_tag = MIR
		original_tag = RIV
		original_tag = HLI
	}
}

is_dwarves = {
	OR = {
		original_tag = ERE
		AND = {
			original_tag = MOA
			NOT = { has_government = belligerent }
		}
	}
}

is_orcs = {
	OR = {
		original_tag = MOR
		original_tag = ISE
		original_tag = ANG
		original_tag = MMO
		original_tag = DGU
		AND = {
			original_tag = MOA
			has_government = belligerent
		}
	}
}

is_hobbits = {
	OR = {
		original_tag = SHI
	}
}

is_men = {
	NOT = {
		is_elves = yes
		is_orcs = yes
		is_hobbits = yes
		is_dwarves = yes
	}
}
