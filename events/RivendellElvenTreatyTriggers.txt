###########################
# Rivendell Events
###########################

add_namespace = rivendell

country_event = {
	id = rivendell.1000
	hidden = yes
	
	fire_only_once = yes

	trigger = {
		has_global_flag = elven_treaty_lindon
		has_global_flag = elven_treaty_lorien
	}
	
	immediate = {
		if = {
			limit = { NOT = { LIT = { has_war_with = LIN } } }
			LIT = { diplomatic_relation = { country = LIN relation = non_aggression_pact } }
			LIN = { diplomatic_relation = { country = LIT relation = non_aggression_pact } }
		}
	}
	
	option = {
		name = rivendell.1000.a
	}
}
