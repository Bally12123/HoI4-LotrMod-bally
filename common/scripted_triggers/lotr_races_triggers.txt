is_orcs = {
	OR = {
		original_tag = MOR
		original_tag = MMO
		original_tag = DGU
		original_tag = MOC
		AND = {
			original_tag = ISE
			has_country_flag = has_switched_to_orcs
		}
		AND = {
			original_tag = ANG
			has_country_flag = has_switched_to_orcs
		}
	}
}
