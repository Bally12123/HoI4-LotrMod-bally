add_namespace = arthedain


# Arthedain asks about Mutual Defense Pact
country_event = {
	id = arthedain.1
	title = arthedain.1.t
	desc = arthedain.1.d
	picture = GFX_report_event_paper
	
	is_triggered_only = yes
	
	option = { #Accept
		name = arthedain.1.a
		ai_chance = {
			base = 60
		}
		ARG = { give_guarantee = PREV }
		THIS = { give_guarantee = ARG }
	}
	option = { #Refuse
		name = arthedain.1.b
		ai_chance = {
			base = 50
			modifier = {
				add = 5000
				has_war_with = ARG
			}
		}
	}
}

# Demand Cardolan / Rhudaur
country_event = {
	id = arthedain.2
	title = arthedain.2.t
	desc = arthedain.2.d
	picture = GFX_report_event_paper

	is_triggered_only = yes

	option = { # Accept
		name = arthedain.2.a
		ai_chance = {
			base = 25
			modifier = {
				add = 10
				THIS = { has_government = fascism }
			}
			modifier = {
				has_opinion = { target = ARG value > 50 }
				add = 30
			}
		}
 		ARG = { annex_country = { target = PREV } }
	}
	option = { # Refuse
		name = arthedain.2.b
		ai_chance = {
			base = 50
			modifier = {
				any_other_country = {
					has_guaranteed = THIS
				}
				factor = 5
			}
			modifier = {
				add = 30
				strength_ratio = { tag = ARG ratio > 0.5 }
			}
			modifier = {
				add = 100
				is_in_faction = yes
			}
			modifier = {
				add = 500
				strength_ratio = { tag = ARG ratio > 1.0 }
			}
		}
		ARG = { create_wargoal = { type = annex_everything target = PREV }  }
	}
}
