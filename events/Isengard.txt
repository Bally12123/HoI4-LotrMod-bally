# Saruman Demands Enedwaith Total Submission
country_event = {
	id = isengard.4
	title = isengard.4.t
	desc = isengard.4.d
	picture = GFX_report_event_paper

	is_triggered_only = yes

	option = { # Accept
		name = isengard.4.a
		ai_chance = {
			base = 30
			modifier = {
				add = -25
				ISR = { has_army_size = { size < 30 } }
			}
			modifier = {
				add = 30
				ISR = { has_army_size = { size > 40 } }
			}
			modifier = {
				add = 100
				EGY = { has_government = fascism }
			}
		}
 		ISR = { annex_country = { target = EGY } }
	}
	option = { # Refuse
		name = isengard.4.b
		ai_chance = {
			base = 50
			modifier = {
				any_other_country = {
					has_guaranteed = EGY
				}
				factor = 5
			}
			modifier = {
				add = 20
				EGY = { has_army_size = { size > 20 } }
			}
			modifier = {
				factor = 0
				is_historical_focus_on = yes
			}
		}
		EGY = { declare_war_on = { target = ISR type = annex_everything } }
	}
}
