


# checks if THIS has the same race as ROOT
is_same_race = {
	OR = {
		AND = { ROOT = { is_men = yes } is_men = yes }
		AND = { ROOT = { is_elves = yes } is_elves = yes }
		AND = { ROOT = { is_orcs = yes } is_orcs = yes }
		AND = { ROOT = { is_dwarves= yes } is_dwarves = yes }
	}
}
