# Communists Execute Coup (Angmar only)
country_event = {
	id = angmar.2
	title = political.3.t
	desc = political.3.d
	picture = GFX_report_event_gathering_fight
	
	trigger = {
		NOT = { has_government = revolutionary }
		revolutionary > 0.4
		is_major = no
		tag = ANG
		has_global_flag = ang_terrorism
	}

	mean_time_to_happen = {
		days = 100
		modifier = {
			factor = 0.5
			revolutionary > 0.6
		}
	}

	immediate = {
	}
	
	option = {
		name = political.3.a
		set_politics = {
			ruling_party = revolutionary
			elections_allowed = no
		}
		add_timed_idea = { idea = reign_of_terror days = 730 }
	}
}

#democrats take over
country_event = {
	id = angmar.4
	title = political.17.t
	desc = political.17.d
	picture = GFX_report_event_election_vote
	
	trigger = {
		NOT = { has_government = cooperative }
		cooperative > 0.4
		is_puppet = no
		tag = ANG
	}
	
	mean_time_to_happen = {
		days = 120
	}
	
	option = {
		name = political.17.a
		if = {
			limit = { has_government = belligerent }
			add_timed_idea = { idea = belligerent_defeated days = 730 }
		}
		if = {
			limit = { has_government = revolutionary }
			add_timed_idea = { idea = revolutionary_defeated days = 730 }
		}
		set_politics = {
			ruling_party = cooperative
			elections_allowed = no
		}
	}
	option = {
		name = political.17.b
	}
}
